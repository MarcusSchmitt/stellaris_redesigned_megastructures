dyson_sphere_1 = {
	entity = "dyson_sphere_phase_01_entity"
	construction_entity = "dyson_sphere_frame_entity"
	portrait = "GFX_megastructure_construction_background"
	upgrade_from = {
		dyson_sphere_0
	}
	build_time = @frame_build_time
	build_cost = {
		minerals = @frame_build_minerals
	}
	construction_blocks_others = @block_other_construction

	possible = {
		from = { has_ascension_perk = ap_galactic_wonders }
	}
	on_build_start = {
		every_system_planet = { 
			limit = { 
				AND = {
					is_star = no
					NOT ={is_planet_class = pc_habitat}
				}
			}
			add_modifier = {
				modifier = "death_by_megastructure"
				days = @frame_fear_time
			} 
		}
	}
	on_build_cancel = {}
	on_build_complete = {
		set_star_flag = dyson_sphere_frame_built
		every_system_planet = { 
			limit = { 
				NOR = {
					is_star = yes
					is_asteroid = yes
				}
			}
			set_planet_flag = dyson_sphere_frame_built
			add_modifier = {
				modifier = "dyson_sphere_frame_reduced_sun"
				days = -1
			} 
		}
	}
}

dyson_sphere_5 = {
	entity = "dyson_sphere_phase_05_entity"
	construction_entity = "dyson_sphere_part_4_entity"
	portrait = "GFX_megastructure_dyson_sphere_background"
	upgrade_from = {
		dyson_sphere_4
	}
	build_time = @fill_frame_build_time
	build_cost = {
		minerals = @fill_frame_minerals
	}
	monthly_production = {
		energy = @panel_4_energy_production
	}

	possible = {
		from = { has_ascension_perk = ap_galactic_wonders }
	}

	construction_blocks_others = @block_other_construction
	on_build_complete = {
		if {
			limit = {
				any_planet = {
					OR = {
						is_colony = yes
						is_under_colonization = yes 
					}
					NOT = {
						has_planet_flag = megastructure
					}
				}
			}
			
			every_country = {
				limit = {
					has_communications = from
					NOT = { is_country_type = primitive }
					NOT = { is_country = from }
				}
				add_opinion_modifier = { who = from modifier = opinion_sphere_genocidal }
			}
		}
		every_system_planet = {
			limit = {
				OR = {
					is_planet_class = pc_molten
					is_planet_class = pc_toxic
					is_planet_class = pc_nuked
				}
			}
			change_pc = pc_frozen
		}
		every_system_planet = {
			limit = {
				is_planet_class = pc_barren
			}
			change_pc = pc_barren_cold
		}
		every_system_planet = {
			limit = {
				OR = {
					is_planet_class = pc_desert
					is_planet_class = pc_arid
					is_planet_class = pc_savannah
					is_planet_class = pc_tropical
					is_planet_class = pc_continental
					is_planet_class = pc_ocean
					is_planet_class = pc_arctic
					is_planet_class = pc_tundra
					is_planet_class = pc_alpine
					is_planet_class = pc_gaia
				}
			}
			if = {
				limit = {
					OR = {
						is_colony = yes
						is_under_colonization = yes 
					}
				}
				destroy_colony = { keep_buildings = no }
			}
			change_pc = pc_barren_cold
			add_modifier = {
				modifier = "terraforming_candidate"
				days = -1
			}
		}
		every_system_planet = { 
			limit = { is_star = no }
			remove_modifier = dyson_sphere_panel_3_reduced_sun
			remove_planet_flag = dyson_sphere_panel_3_built
			set_planet_flag = dyson_sphere_panel_4_built
			add_modifier = {
				modifier = "dyson_sphere_panel_4_reduced_sun"
				days = -1
			} 
		}
	}
}
